#include <GL/glut.h>

// Draw text labels
void drawText(float x, float y, const char* text) {
    glRasterPos2f(x, y);
    while (*text) {
        glutBitmapCharacter(GLUT_BITMAP_HELVETICA_12, *text);
        text++;
    }
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    // ================= GL_POINTS =================
    glPointSize(5);
    glBegin(GL_POINTS);
    glVertex2f(-0.9, 0.8);
    glVertex2f(-0.8, 0.7);
    glVertex2f(-0.7, 0.75);
    glVertex2f(-0.6, 0.85);
    glEnd();
    drawText(-0.9, 0.65, "GL_POINTS");

    // ================= GL_LINES =================
    glBegin(GL_LINES);
    glVertex2f(-0.9, 0.4); glVertex2f(-0.7, 0.5);
    glVertex2f(-0.8, 0.3); glVertex2f(-0.6, 0.4);
    glVertex2f(-0.9, 0.3); glVertex2f(-0.7, 0.2);
    glEnd();
    drawText(-0.9, 0.15, "GL_LINES");

    // ================= GL_LINE_STRIP =================
    glBegin(GL_LINE_STRIP);
    glVertex2f(-0.2, 0.4);
    glVertex2f(0.0, 0.5);
    glVertex2f(0.2, 0.4);
    glVertex2f(0.0, 0.3);
    glEnd();
    drawText(-0.1, 0.2, "GL_LINE_STRIP");

    // ================= GL_LINE_LOOP =================
    glBegin(GL_LINE_LOOP);
    glVertex2f(0.6, 0.4);
    glVertex2f(0.8, 0.5);
    glVertex2f(0.9, 0.3);
    glVertex2f(0.7, 0.2);
    glEnd();
    drawText(0.6, 0.15, "GL_LINE_LOOP");

    // ================= GL_TRIANGLES =================
    glBegin(GL_TRIANGLES);
    glVertex2f(-0.9, -0.1);
    glVertex2f(-0.8, 0.1);
    glVertex2f(-0.7, -0.1);

    glVertex2f(-0.6, -0.1);
    glVertex2f(-0.5, 0.1);
    glVertex2f(-0.4, -0.1);
    glEnd();
    drawText(-0.9, -0.2, "GL_TRIANGLES");

    // ================= GL_TRIANGLE_STRIP =================
    glBegin(GL_TRIANGLE_STRIP);
    glVertex2f(-0.2, -0.1);
    glVertex2f(-0.2, 0.1);
    glVertex2f(0.0, -0.1);
    glVertex2f(0.0, 0.1);
    glVertex2f(0.2, -0.1);
    glVertex2f(0.2, 0.1);
    glEnd();
    drawText(-0.2, -0.2, "GL_TRIANGLE_STRIP");

    // ================= GL_TRIANGLE_FAN =================
    glBegin(GL_TRIANGLE_FAN);
    glVertex2f(0.6, 0.0);
    glVertex2f(0.7, 0.2);
    glVertex2f(0.8, 0.1);
    glVertex2f(0.9, 0.0);
    glVertex2f(0.8, -0.1);
    glEnd();
    drawText(0.6, -0.2, "GL_TRIANGLE_FAN");

    // ================= GL_QUADS =================
    glBegin(GL_QUADS);
    glVertex2f(-0.9, -0.5);
    glVertex2f(-0.7, -0.5);
    glVertex2f(-0.7, -0.7);
    glVertex2f(-0.9, -0.7);
    glEnd();
    drawText(-0.9, -0.8, "GL_QUADS");

    // ================= GL_QUAD_STRIP =================
    glBegin(GL_QUAD_STRIP);
    glVertex2f(-0.2, -0.5);
    glVertex2f(-0.2, -0.7);
    glVertex2f(0.0, -0.5);
    glVertex2f(0.0, -0.7);
    glVertex2f(0.2, -0.5);
    glVertex2f(0.2, -0.7);
    glEnd();
    drawText(-0.2, -0.8, "GL_QUAD_STRIP");

    // ================= GL_POLYGON =================
    glBegin(GL_POLYGON);
    glVertex2f(0.6, -0.5);
    glVertex2f(0.8, -0.5);
    glVertex2f(0.9, -0.7);
    glVertex2f(0.7, -0.8);
    glVertex2f(0.5, -0.7);
    glEnd();
    drawText(0.6, -0.9, "GL_POLYGON");

    glFlush();
}

void init() {
    glClearColor(1, 1, 1, 1);
    glColor3f(0, 0, 0);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutCreateWindow("OpenGL Primitives");
    glutInitWindowSize(800, 600);
    glutDisplayFunc(display);
    init();
    glutMainLoop();
}
